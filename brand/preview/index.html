app.ts
import { Client } from '@itellico/sdk';

// Initialize the client
const client = new Client(process.env.API_KEY);

// Start a compliant voice session
const session = await client.voice.create({
  to: '+491510000000',
  prompt: 'Appointment scheduling voice AI teammate...',
  locale: 'de-DE',
  compliance_badge: true,
  record: true
});

// Log the session ID for audit
console.log(`Session started: ${session.id}`);
